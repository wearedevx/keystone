# Init project

ks init test-project

! grep 'label=value' .keystone/cache/dev/.env
! grep 'label=value' .keystone/cache/staging/.env

grep 'label=value' .keystone/cache/dev/.env
grep 'label=value' .keystone/cache/staging/.env
grep 'label=value' .keystone/cache/prod/.env
